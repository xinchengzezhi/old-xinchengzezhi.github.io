1.1.3 事务
 事务管理器
Hibernate –》 HibernateTransactionManager
JDBC –> DataSourceTransactionManager
 事务详情<tx:advice id=”” transaction-manager=””>
tx:attribute <tx:method=”” 传播行为、隔离级别、是否只读、异常>
 aop编程，事务通知应用到目标类aop:config aop:advisor





<!– 3.2 事务详情 

    * 增删改：读写；查询：只读
-->
<tx:advice id="txAdvice" transaction-manager="txManager">
    <tx:attributes>
        <tx:method name="add*"/>
        <tx:method name="update*"/>
        <tx:method name="delete*"/>
        <tx:method name="login" read-only="true"/>
        <tx:method name="find*" read-only="true"/>
    </tx:attributes>
</tx:advice>
<!-- 3.3 aop编程  
    * 强制cglib ：  proxy-target-class="true"
-->
<aop:config>
    <aop:advisor advice-ref="txAdvice" pointcut="execution(* com.itheima.crm.*.service..*.*(..))"/>
</aop:config>
